--Switch Configuration Cheat sheet 

enable
configure terminal        #RENAME THE SWITCH TO S1
hostname S1

interface gi0/1
swicthport mode trunk     #SWITCH PORT GI0/1 TO TRUNK
swicthport nonegotiate
exit

interface gi0/2
swicthport mode trunk     #SWITCH PORT GI0/2 TO TRUNK 
swicthport nonegotiate
exit

vlan 3                   #CREATE VLAN 3 AND NAME IT Management
name Management

interface vlan 3
ip address 192.168.3.2 255.255.255.0
no shutdown
exit                     # ASSIGN A IP AND SUBNET MASK AND CHANGE THE STATE TO UP

ip default-gateway 192.168.3.1    # GIVES THE SWITCH A DEFAULT GATEWAY

vlan 10                  #CREATE VLAN 10 AND NAME IT TEAM1
name Team1
exit

vlan 20                  #CREATE VLAN 20 AND NAME IT TEAM2
name Team2
exit

line vty 0 15            #CREATES A PASSWORD FOR LINE 0-15 FOR LOGIN/REMOTE ACCESS/TELNET
password cisco
login
transport input telnet
exit

interface fa0/1          #SETS PORT FA0/1 TO VLAN 10
switchport mode access
switchport access vlan 10
exit

interface fa0/13         #SETS PORT FA0/13 TO VLAN 10
switchport mode access
swicthport access vlan 10
exit

--InterVLAN Router CONFIG------------------------------------------

no shutdown             #ASSIGNS PORT GO0/0 A IP ADDRESS AND TURNS STATE TO UP
exit

no shutdown
exit

no shutdown
exit

--STATIC ROUTING----------------------------------------------------

enable
conf t                                     
ip route 10.0.2.0 255.255.255.0 10.0.7.2   #DESTINATION-----SUBNETMASK----NEXT ROUTER(MATE)

router rip -> redistribute static ###


--DYNAMIC ROUTING-----------------------------------------------------

enable 
conf t
router rip
version 2                      #CONFIGURATION OF RIP (INTERIER PROTOCOL
network 10.0.0.0
no auto-summary
exit


ip dhcp excluded-address 192.168.3.1 192.168.3.10     #EXLUDED IP ADDRESS FROM 3.1 - 3.10

ip dhcp pool 'nameCOMPUTING'
network 192.168.3.0 255.255.255.0    #AUTO HAND OUT OF IP ADDRESS TO PC
default-router 192.168.3.1
exit
